<script>
    /* --- MAGIC CURSOR LOGIC --- */
    const navContainer = document.getElementById('navContainer');
    const highlight = document.getElementById('highlight');
    const btns = document.querySelectorAll('.nav-btn');

    function updateHighlight(targetBtn) {
        const btnRect = targetBtn.getBoundingClientRect();
        const containerRect = navContainer.getBoundingClientRect();
        
        // Calculate relative position
        const top = btnRect.top - containerRect.top;
        
        highlight.style.opacity = '1';
        highlight.style.top = `${top}px`;
        highlight.style.height = `${btnRect.height}px`;
    }

    // Initialize highlight on active button
    const activeBtn = document.querySelector('.nav-btn.active');
    if (activeBtn) setTimeout(() => updateHighlight(activeBtn), 100); // Small delay to ensure render

    // Add Hover Listener to all buttons
    btns.forEach(btn => {
        btn.addEventListener('mouseenter', () => updateHighlight(btn));
        btn.addEventListener('click', () => {
             btns.forEach(b => b.classList.remove('active'));
             btn.classList.add('active');
        });
    });

    // Reset to active button when mouse leaves container
    navContainer.addEventListener('mouseleave', () => {
        const currentActive = document.querySelector('.nav-btn.active');
        if (currentActive) updateHighlight(currentActive);
    });


    /* --- PAGE SWITCHING --- */
    function switchPage(pageId, btn) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
        document.getElementById(pageId).classList.add('active');
    }

    /* --- UTILITIES --- */
    function copyKey(el) {
        const original = el.innerHTML;
        navigator.clipboard.writeText("sk_live_51MzXy2...");
        
        el.style.borderColor = "var(--cyan)";
        el.innerHTML = `<span style="color:var(--cyan); filter:blur(0);">Copied!</span>`;
        
        setTimeout(() => {
            el.innerHTML = original;
            el.style.borderColor = "";
        }, 2000);
    }

    // Trigger progress bar animations on load
    window.onload = () => {
        // Redraw widths to trigger CSS transitions
        const bars = document.querySelectorAll('.progress-fill');
        bars.forEach(bar => {
            const width = bar.style.width;
            bar.style.width = '0';
            setTimeout(() => bar.style.width = width, 300);
        });
    };
</script>
